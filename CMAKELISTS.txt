
cmake_minimum_required(VERSION 3.10)

project(pwshbuild)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

add_compile_definitions(DEBUG)

add_executable(pwshbuild
	./src/pwshbuild/main.cpp

	./src/pwshbuild/cli/generator.h
	./src/pwshbuild/cli/cliparser.h
	./src/pwshbuild/cli/generator.cpp
	./src/pwshbuild/cli/cliparser.cpp

	./src/pwshbuild/utils/primitives.h
)


target_include_directories(pwshbuild PUBLIC ./src)
